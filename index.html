步驟一：建立 Google 試算表

1. 開啟 Google Sheets，新增試算表
2. 將第一個工作表命名為 申請記錄
3. 在 A1 開始依序填入欄位標題：
申請ID | 單位 | 姓名 | 裝備 | 規格 | 損壞描述 | 汰換原因 | 狀態 | 管理員備註 | 申請時間 | 審核時間
4. 再新增第二個工作表，命名為 庫存
5. 在 A1 填入：裝備ID | 裝備名稱 | 庫存數量 | 最後更新

步驟二：建立 Apps Script

1. 在試算表中點選「擴充功能」→「Apps Script」
2. 刪除預設程式碼，貼上以下 Script：
const SS_ID = SpreadsheetApp.getActive().getId(); function doPost(e) { const data = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.openById(SS_ID); const action = data.action; if (action === 'addRequest') { const sh = ss.getSheetByName('申請記錄'); sh.appendRow([ data.id, data.unit, data.name, data.equipName, data.specs, data.condition, data.reason, '待審核', '', data.submittedAt, '' ]); } if (action === 'updateRequest') { const sh = ss.getSheetByName('申請記錄'); const rows = sh.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (rows[i][0] === data.id) { sh.getRange(i+1, 8).setValue(data.status); sh.getRange(i+1, 9).setValue(data.note || ''); sh.getRange(i+1, 11).setValue(new Date().toLocaleString('zh-TW')); break; } } } if (action === 'updateStock') { const sh = ss.getSheetByName('庫存'); const rows = sh.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (rows[i][0] === data.id) { sh.getRange(i+1, 3).setValue(data.stock); sh.getRange(i+1, 4).setValue(new Date().toLocaleString('zh-TW')); break; } } } return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON); } function doGet(e) { const ss = SpreadsheetApp.openById(SS_ID); const action = e.parameter.action; if (action === 'ping') { return ContentService.createTextOutput(JSON.stringify({ok:true,msg:'連線成功'})).setMimeType(ContentService.MimeType.JSON); } if (action === 'getAll') { const reqSh = ss.getSheetByName('申請記錄'); const invSh = ss.getSheetByName('庫存'); const reqRows = reqSh.getDataRange().getValues().slice(1); const invRows = invSh.getDataRange().getValues().slice(1); const requests = reqRows.filter(r=>r[0]).map(r => ({ id:r[0],unit:r[1],name:r[2],equipName:r[3], specsStr:r[4],condition:r[5],reason:r[6], status:r[7]||'pending',adminNote:r[8]||'',submittedAt:r[9] })); const inventory = invRows.filter(r=>r[0]).map(r => ({ id:r[0],name:r[1],stock:Number(r[2]),updated:r[3] })); return ContentService.createTextOutput(JSON.stringify({requests,inventory})).setMimeType(ContentService.MimeType.JSON); } return ContentService.createTextOutput(JSON.stringify({ok:false})).setMimeType(ContentService.MimeType.JSON); }

步驟三：部署為網頁應用程式

1. 點右上角「部署」→「新增部署作業」
2. 類型選「網頁應用程式」
3. 執行身份選「我」，存取權選「所有人」
4. 點「部署」，複製產生的網址貼到上方設定欄
⚠️ 每次修改 Script 後需重新部署（管理現有部署 → 編輯 → 版本選「新版本」），才會生效。
