🔗 Google Sheets 連線設定
設定後所有申請即時寫入 Google Sheets，管理員可從雲端同步讀取所有人的資料。
⚠️ 解決「看不到別人申請」的根本方法：申請資料預設只存在填表者的瀏覽器（localStorage）。設定 Sheets 後，每次提交都寫入雲端，承辦人點「立即從雲端同步」即可看到所有人的申請。
📘 快速設定步驟
1. 建立 Google 試算表，新增兩個工作表：申請記錄 與 庫存
2. 在試算表點「擴充功能」→「Apps Script」，貼上下方程式碼並儲存
3. 點「部署」→「新增部署作業」→ 類型選「網頁應用程式」→ 存取權選「所有人」
4. 複製部署網址貼到上方，點「儲存網址」再「測試連線」
const SS = SpreadsheetApp.getActive(); function doPost(e){ const d=JSON.parse(e.postData.contents),ss=SS; if(d.action==='addRequest') ss.getSheetByName('申請記錄').appendRow([d.id,d.unit,d.name,d.equipName,d.specs,d.condition,d.reason,'待審核','',d.submittedAt,'']); if(d.action==='updateRequest'){const sh=ss.getSheetByName('申請記錄'),rows=sh.getDataRange().getValues();for(let i=1;i<rows.length;i++){if(String(rows[i][0])===String(d.id)){sh.getRange(i+1,8).setValue(d.status==='approved'?'已核准':'未核准');sh.getRange(i+1,9).setValue(d.note||'');sh.getRange(i+1,11).setValue(new Date().toLocaleString('zh-TW'));break;}}} if(d.action==='updateStock'){const sh=ss.getSheetByName('庫存'),rows=sh.getDataRange().getValues();let f=false;for(let i=1;i<rows.length;i++){if(String(rows[i][0])===String(d.specId)){sh.getRange(i+1,4).setValue(d.stock);sh.getRange(i+1,5).setValue(new Date().toLocaleString('zh-TW'));f=true;break;}}if(!f)sh.appendRow([d.specId,d.equipName,d.spec,d.stock,new Date().toLocaleString('zh-TW')]);} return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON); } function doGet(e){ const ss=SS,a=e.parameter.action; if(a==='ping') return out({ok:true}); if(a==='getAll'){ const toS=s=>s==='已核准'?'approved':s==='未核准'?'rejected':'pending'; const rr=ss.getSheetByName('申請記錄').getDataRange().getValues().slice(1); const ir=ss.getSheetByName('庫存').getDataRange().getValues().slice(1); return out({requests:rr.filter(r=>r[0]).map(r=>({id:String(r[0]),unit:r[1],name:r[2],equipName:r[3],specs:r[4],condition:r[5],reason:r[6],status:toS(r[7]),adminNote:r[8]||'',submittedAt:r[9],photoUrls:r[10]||''})),stockRows:ir.filter(r=>r[0]).map(r=>({specId:String(r[0]),equipName:r[1],spec:r[2],stock:Number(r[3]),updated:r[4]}))}); } return out({ok:false}); } function out(o){return ContentService.createTextOutput(JSON.stringify(o)).setMimeType(ContentService.MimeType.JSON);}
