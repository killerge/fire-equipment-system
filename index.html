// ══ 第五大隊裝備汰換系統 Apps Script v4 ══ const SS = SpreadsheetApp.getActive(); function doPost(e) { const d = JSON.parse(e.postData.contents); if (d.action === 'addRequest') { ensureSheet(); SS.getSheetByName('申請記錄').appendRow([ d.id, d.unit, d.name, d.equipName, d.specs, d.condition, d.reason, '待審核', '', d.submittedAt, d.photoUrls || '' ]); } if (d.action === 'uploadPhoto') { try { const bytes = Utilities.base64Decode(d.b64); const blob = Utilities.newBlob(bytes, d.mime, d.name); const folder = getFolder(); const file = folder.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); // 使用 thumbnail URL，各瀏覽器皆可顯示 const url = 'https://drive.google.com/thumbnail?id=' + file.getId() + '&sz=w400'; return out({ ok: true, url: url, id: file.getId() }); } catch(err) { return out({ ok: false, error: err.toString() }); } } if (d.action === 'updateRequest') { const sh = SS.getSheetByName('申請記錄'); const rows = sh.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (String(rows[i][0]) === String(d.id)) { sh.getRange(i+1,8).setValue(d.status === 'approved' ? '已核准' : '未核准'); sh.getRange(i+1,9).setValue(d.note || ''); sh.getRange(i+1,11).setValue(new Date().toLocaleString('zh-TW')); break; } } } if (d.action === 'updateStock') { const sh = SS.getSheetByName('庫存'); const rows = sh.getDataRange().getValues(); let found = false; for (let i = 1; i < rows.length; i++) { if (String(rows[i][0]) === String(d.specId)) { sh.getRange(i+1,4).setValue(d.stock); sh.getRange(i+1,5).setValue(new Date().toLocaleString('zh-TW')); found = true; break; } } if (!found) { sh.appendRow([d.specId, d.equipName, d.spec, d.stock, new Date().toLocaleString('zh-TW')]); } } return out({ ok: true }); } function doGet(e) { const a = e.parameter.action; if (a === 'ping') return out({ ok: true, v: 4 }); if (a === 'getAll') { ensureSheet(); const toS = s => s==='已核准'?'approved':s==='未核准'?'rejected':'pending'; const rr = SS.getSheetByName('申請記錄').getDataRange().getValues().slice(1); const ir = SS.getSheetByName('庫存').getDataRange().getValues().slice(1); return out({ requests: rr.filter(r=>r[0]).map(r=>({ id: String(r[0]), unit:r[1], name:r[2], equipName:r[3], specs:r[4], condition:r[5], reason:r[6], status:toS(r[7]), adminNote:r[8]||'', submittedAt:String(r[9]||''), photoUrls:String(r[10]||'') })), stockRows: ir.filter(r=>r[0]).map(r=>({ specId:String(r[0]), equipName:r[1], spec:r[2], stock:Number(r[3])||0, updated:String(r[4]||'') })) }); } return out({ ok: false }); } function getFolder() { const name = '消防裝備汰換照片'; const it = DriveApp.getFoldersByName(name); if (it.hasNext()) return it.next(); return DriveApp.createFolder(name); } function ensureSheet() { const names = ['申請記錄','庫存']; const headers = [ ['申請ID','單位','姓名','裝備','規格','損壞描述','汰換原因','狀態','承辦人備註','申請時間','審核時間/照片連結'], ['規格ID','裝備名稱','規格','庫存數量','最後更新'] ]; names.forEach((n,i)=>{ if (!SS.getSheetByName(n)) { const sh = SS.insertSheet(n); sh.getRange(1,1,1,headers[i].length).setValues([headers[i]]); } }); } function out(o) { return ContentService.createTextOutput(JSON.stringify(o)).setMimeType(ContentService.MimeType.JSON); }
