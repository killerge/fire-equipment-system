⚠️ 照片上傳採用 Google Drive 存放（透過 Apps Script 中轉），不需要 Imgur 帳號。上傳後會自動設為公開可檢視，並將連結存入試算表。
// ══ 第五大隊裝備汰換系統 Apps Script v3 ══ // 請完整取代舊程式碼，部署後選「新版本」 const SS = SpreadsheetApp.getActive(); // Google Drive 存放照片的資料夾 ID（留空則自動建立） let PHOTO_FOLDER_ID = ''; function getPhotoFolder() { if (PHOTO_FOLDER_ID) return DriveApp.getFolderById(PHOTO_FOLDER_ID); const folders = DriveApp.getFoldersByName('消防裝備汰換申請照片'); if (folders.hasNext()) { const f = folders.next(); PHOTO_FOLDER_ID = f.getId(); return f; } const f = DriveApp.createFolder('消防裝備汰換申請照片'); PHOTO_FOLDER_ID = f.getId(); return f; } function doPost(e) { const d = JSON.parse(e.postData.contents); const action = d.action; if (action === 'addRequest') { SS.getSheetByName('申請記錄').appendRow([ d.id, d.unit, d.name, d.equipName, d.specs, d.condition, d.reason, '待審核', '', d.submittedAt, d.photoUrls || '' ]); } if (action === 'uploadPhoto') { try { const bytes = Utilities.base64Decode(d.base64); const blob = Utilities.newBlob(bytes, d.mimeType, d.fileName); const file = getPhotoFolder().createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); const url = 'https://drive.google.com/uc?export=view&id=' + file.getId(); return out({ ok: true, url }); } catch (err) { return out({ ok: false, error: err.toString() }); } } if (action === 'updateRequest') { const sh = SS.getSheetByName('申請記錄'); const rows = sh.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (String(rows[i][0]) === String(d.id)) { sh.getRange(i+1, 8).setValue(d.status === 'approved' ? '已核准' : '未核准'); sh.getRange(i+1, 9).setValue(d.note || ''); sh.getRange(i+1, 11).setValue(new Date().toLocaleString('zh-TW')); break; } } } if (action === 'updateStock') { const sh = SS.getSheetByName('庫存'); const rows = sh.getDataRange().getValues(); let found = false; for (let i = 1; i < rows.length; i++) { if (String(rows[i][0]) === String(d.specId)) { sh.getRange(i+1, 4).setValue(d.stock); sh.getRange(i+1, 5).setValue(new Date().toLocaleString('zh-TW')); found = true; break; } } if (!found) { sh.appendRow([d.specId, d.equipName, d.spec, d.stock, new Date().toLocaleString('zh-TW')]); } } return out({ ok: true }); } function doGet(e) { const a = e.parameter.action; if (a === 'ping') return out({ ok: true }); if (a === 'getAll') { const toStatus = s => s === '已核准' ? 'approved' : s === '未核准' ? 'rejected' : 'pending'; const rr = SS.getSheetByName('申請記錄').getDataRange().getValues().slice(1); const ir = SS.getSheetByName('庫存').getDataRange().getValues().slice(1); return out({ requests: rr.filter(r => r[0]).map(r => ({ id: String(r[0]), unit: r[1], name: r[2], equipName: r[3], specs: r[4], condition: r[5], reason: r[6], status: toStatus(r[7]), adminNote: r[8] || '', submittedAt: r[9], photoUrls: r[10] || '' })), stockRows: ir.filter(r => r[0]).map(r => ({ specId: String(r[0]), equipName: r[1], spec: r[2], stock: Number(r[3]), updated: r[4] })) }); } return out({ ok: false }); } function out(o) { return ContentService.createTextOutput(JSON.stringify(o)).setMimeType(ContentService.MimeType.JSON); }
